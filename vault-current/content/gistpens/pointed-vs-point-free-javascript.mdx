  R.unapply(editorCursorMoveAction),
  [selectSelectionStart, selectSelectionEnd],
);

```js title="pointed-action.js"
const elementToCursorMoveAction = (e) =>
  editorCursorMoveAction([selectSelectionStart(e), selectSelectionEnd(e)]);
```
