```js title="pipe.js"
const pipe =
  (...funcs) =>
  (...args) =>
    funcs.reduce((args, func) => [func(...args)], args)[0];
```

```js title="compose.js"
// just reverse the funcs array...?
const compose =
  (...funcs) =>
  (...args) =>
    funcs.reverse().reduce((args, func) => [func(...args)], args)[0];
```
