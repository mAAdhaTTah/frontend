```js title="example.js"
import { query, addClass, append, on } from "pipe-dom";

query("ul")
  |> addClass("important-list")
  |> append(
    document.createElement("li")
      |> addClass("important-item")
      |> on("click", () => console.log("item clicked")),
  );
```

^example-js
